// 泛型
// 把类型的定义推迟到使用时，在调用时传入具体类型
function fne<T>(a: T): T {
  // 三个T都表示同一种类型
  return a;
}
fne(1);
fne("1");
fne(true);

function fne2<A>(a: A) {
  return a;
}
fne2(1); // 推断为number
fne2<string>("1"); // 指定为string
fne2<boolean>(true);
// console.log(fne2<string>(true));
function fmn<X, Y>(a: X, b: Y) {
  return [a, b];
}
